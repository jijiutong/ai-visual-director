**Architecture:** Three-layer local app — Obsidian Vault as data store, Node.js local server as API proxy + workflow engine, vanilla HTML/JS Web UI served at localhost. No database, no deployment. Started via Claude Code Skill command "启动画布".

**Tech Stack:** Node.js (Express + ws), vanilla HTML/CSS/JS (generated by frontend-design skill), Obsidian Vault filesystem

    if (client.readyState === 1) client.send(msg);

  console.log(`AI Story Studio running at http://localhost:${PORT}`);

### Task 3: Vault Reader Module

**Files:**
- Create: `studio/server/vault-reader.js`

- [ ] **Step 1: Create vault-reader.js**

```javascript
const fs = require('fs');
const path = require('path');
const matter = require('gray-matter');

class VaultReader {
  constructor(vaultPath) {
    this.vaultPath = vaultPath || process.env.OBSIDIAN_VAULT || process.cwd();
  }

  /**
   * Scan a directory and return its tree structure.
   * @param {string} relativePath - relative path within vault
   * @returns {{ name: string, type: 'file'|'dir', path: string, children?: array }}
   */
  scanDir(relativePath = '') {
    const fullPath = path.join(this.vaultPath, relativePath);
    const entries = fs.readdirSync(fullPath, { withFileTypes: true });

    return entries
      .filter(e => !e.name.startsWith('.') && !e.name.startsWith('_'))
      .sort((a, b) => {
        if (a.isDirectory() && !b.isDirectory()) return -1;
        if (!a.isDirectory() && b.isDirectory()) return 1;
        return a.name.localeCompare(b.name);
      })
      .map(e => {
        const entryPath = relativePath ? `${relativePath}/${e.name}` : e.name;
        if (e.isDirectory()) {
          return {
            name: e.name,
            type: 'dir',
            path: entryPath,
            children: this.scanDir(entryPath)
          };
        }
        if (e.name.endsWith('.md')) {
          const stat = fs.statSync(path.join(fullPath, e.name));
          const content = fs.readFileSync(path.join(fullPath, e.name), 'utf-8');
          const charCount = content.replace(/\s/g, '').length;
          return {
            name: e.name.replace('.md', ''),
            type: 'file',
            path: entryPath,
            charCount,
            size: stat.size
          };
        }
        return null;
      })
      .filter(Boolean);
  }

  /**
   * Read a markdown file with frontmatter parsing.
   * @returns {{ content: string, frontmatter: object, raw: string }}
   */
  readFile(relativePath) {
    const fullPath = path.join(this.vaultPath, relativePath);
    const raw = fs.readFileSync(fullPath, 'utf-8');
    const parsed = matter(raw);
    return {
      content: parsed.content,
      frontmatter: parsed.data,
      raw
    };
  }

  /**
   * Find files matching a pattern in the vault.
   */
  findFiles(pattern, baseDir = '') {
    const results = [];
    const fullPath = path.join(this.vaultPath, baseDir);

    function walk(dir, accPath) {
      const entries = fs.readdirSync(dir, { withFileTypes: true });
      for (const e of entries) {
        if (e.name.startsWith('.')) continue;
        const relPath = accPath ? `${accPath}/${e.name}` : e.name;
        if (e.isDirectory()) {
          walk(path.join(dir, e.name), relPath);
        } else if (pattern.test(e.name)) {
          results.push(relPath);
        }
      }
    }

    walk(fullPath, baseDir);
    return results;
  }

  /**
   * Scan a novel directory: find chapters + associated character/scene notes.
   * @returns {{ chapters: array, characters: array, scenes: array }}
   */
  scanProject(projectDir) {
    const chapters = [];
    const characters = [];
    const scenes = [];

    const fullPath = path.join(this.vaultPath, projectDir);
    const entries = fs.readdirSync(fullPath, { withFileTypes: true });

    for (const e of entries) {
      if (e.name.startsWith('.')) continue;
      if (e.isDirectory()) {
        if (e.name === '角色' || e.name === 'characters') {
          const charFiles = fs.readdirSync(path.join(fullPath, e.name))
            .filter(f => f.endsWith('.md'));
          for (const f of charFiles) {
            const data = this.readFile(`${projectDir}/${e.name}/${f}`);
            characters.push({ name: f.replace('.md', ''), path: `${projectDir}/${e.name}/${f}`, frontmatter: data.frontmatter });
          }
        } else if (e.name === '场景' || e.name === 'scenes') {
          const sceneFiles = fs.readdirSync(path.join(fullPath, e.name))
            .filter(f => f.endsWith('.md'));
          for (const f of sceneFiles) {
            const data = this.readFile(`${projectDir}/${e.name}/${f}`);
            scenes.push({ name: f.replace('.md', ''), path: `${projectDir}/${e.name}/${f}`, frontmatter: data.frontmatter });
          }
        }
      } else if (e.name.endsWith('.md') && !e.name.startsWith('_')) {
        const data = this.readFile(`${projectDir}/${e.name}`);
        chapters.push({
          name: e.name.replace('.md', ''),
          path: `${projectDir}/${e.name}`,
          frontmatter: data.frontmatter,
          charCount: data.content.replace(/\s/g, '').length
        });
      }
    }

    chapters.sort((a, b) => a.name.localeCompare(b.name));
    return { chapters, characters, scenes };
  }
}

module.exports = VaultReader;
```

    },
    {

      icon: '🎥',
      flowFile: 'board-to-video'
    },
    {
      name: '一键全来',
      icon: '⚡',
      flowFile: 'one-click-all'
    }
  ],
  custom_workflows: []
};

class ConfigManager {
  constructor(vaultPath) {
    this.vaultPath = vaultPath || process.env.OBSIDIAN_VAULT || process.cwd();
    this.configPath = path.join(this.vaultPath, 'config.yml');
    this.config = null;
  }

  load() {
    if (!fs.existsSync(this.configPath)) {
      this.config = { ...DEFAULT_CONFIG };
      return this.config;
    }
    const raw = fs.readFileSync(this.configPath, 'utf-8');
    const userConfig = yaml.load(raw) || {};
    // Deep merge: user config overrides defaults
    this.config = this._merge(DEFAULT_CONFIG, userConfig);
    return this.config;
  }

  save(newConfig) {
    this.config = this._merge(this.config, newConfig);
    const yamlStr = yaml.dump(this.config, {
      indent: 2,
      lineWidth: 120,
      noRefs: true
    });
    fs.writeFileSync(this.configPath, yamlStr, 'utf-8');
    return this.config;
  }

  get(key) {
    if (!this.config) this.load();
    return key ? this._getNested(this.config, key) : this.config;
  }

  getPlatformKey(platformName) {
    const config = this.get();
    return config.platforms?.[platformName]?.key || config.image_platforms?.[platformName]?.key || null;
  }

  getPresetFlow(flowName) {
    const config = this.get();
    return [...(config.preset_workflows || []), ...(config.custom_workflows || [])]
      .find(w => w.flowFile === flowName || w.name === flowName);
  }

  _merge(base, override) {
    const result = { ...base };
    for (const key of Object.keys(override)) {
      if (override[key] && typeof override[key] === 'object' && !Array.isArray(override[key])) {
        result[key] = this._merge(base[key] || {}, override[key]);
      } else {
        result[key] = override[key];
      }
    }
    return result;
  }

  _getNested(obj, path) {
    return path.split('.').reduce((o, k) => (o || {})[k], obj);
  }
}

module.exports = ConfigManager;

/**
 * API Proxy — proxies requests to external image/video generation APIs.
 * All API keys are loaded from ConfigManager, never exposed to frontend.
 */

const { exec } = require('child_process');
const path = require('path');
const fs = require('fs');

class APIProxy {
  constructor(configManager) {
    this.config = configManager;
  }

  /**
   * Call a video platform API (async, returns task ID).
   */
  async callVideoAPI(platform, params) {
    const key = this.config.getPlatformKey(platform);
    if (!key) throw new Error(`No API key configured for platform: ${platform}`);

    switch (platform) {
      case 'seedance':
        return this._callSeedance(key, params);
      case 'runway':
        return this._callRunway(key, params);
      case 'kling':
        return this._callKling(key, params);
      case 'luma':
        return this._callLuma(key, params);
      case 'pika':
        return this._callPika(key, params);
      default:
        throw new Error(`Unknown video platform: ${platform}`);
    }
  }

  /**
   * Poll video task status until complete.
   * @returns {Promise<{status: string, result?: object}>}
   */
  async pollVideoTask(platform, taskId, maxRetries = 60, interval = 5000) {
    const key = this.config.getPlatformKey(platform);
    for (let i = 0; i < maxRetries; i++) {
      const result = await this._pollTask(platform, key, taskId);
      if (result.status === 'completed' || result.status === 'failed') {
        return result;
      }
      await this._sleep(interval);
    }
    throw new Error(`Task ${taskId} timed out after ${maxRetries * interval}ms`);
  }

  /**
   * Call image platform API (synchronous, returns image URL or base64).
   */
  async callImageAPI(platform, params) {
    const key = this.config.getPlatformKey(platform);
    if (!key) throw new Error(`No API key configured for image platform: ${platform}`);

    switch (platform) {
      case 'gpt_image':
        return this._callDALLE(key, params);
      case 'flux':
        return this._callReplicate(key, 'flux', params);
      case 'ideogram':
        return this._callIdeogram(key, params);
      case 'nano_banana':
        return this._callNanoBanana(key, params);
      default:
        throw new Error(`Unknown image platform: ${platform}`);
    }
  }

  // --- Platform-specific implementations (stubs with real endpoints) ---

  async _callSeedance(key, params) {
    const endpoint = this.config.get('platforms.seedance.endpoint') ||
      'https://ark.cn-beijing.volces.com/api/v1/video/generation';
    const resp = await fetch(endpoint, {
      method: 'POST',
      headers: { 'Authorization': `Bearer ${key}`, 'Content-Type': 'application/json' },
      body: JSON.stringify({
        prompt: params.prompt,
        reference_images: params.referenceImages || [],
        duration: params.duration || 15,
        aspect_ratio: params.aspectRatio || '16:9'
      })
    });
    if (!resp.ok) throw new Error(`Seedance API error: ${await resp.text()}`);
    return (await resp.json()).task_id;
  }

  async _callRunway(key, params) {
    const resp = await fetch('https://api.runwayml.com/v1/generate', {
      method: 'POST',
      headers: { 'Authorization': `Bearer ${key}`, 'Content-Type': 'application/json' },
      body: JSON.stringify({
        prompt: params.prompt,
        image_url: params.imageUrl,
        duration: params.duration || 4,

      })

  async _callNanoBanana(key, params) {
    // Google Gemini API for image generation
    const resp = await fetch(
      `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash-exp:generateContent?key=${key}`,
      {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },

      }
    );
    if (!resp.ok) throw new Error(`Gemini API error: ${await resp.text()}`);
    return await resp.json();
  }

  async _pollTask(platform, key, taskId) {
    // Platform-specific polling endpoints
    const endpoints = {

/**
 * Workflow Engine — parses flow definition JSON, runs topological sort,
 * generates execution plan with parallel groups.
 *
 * Flow JSON format:
 * {
 *   "nodes": [{ "id": "read", "type": "obsidian_read", "label": "读Obsidian", "params": {} }, ...],
 *   "edges": [{ "from": "read", "to": "extract" }, ...]
 * }
 */

class WorkflowEngine {
  /**
   * Parse flow definition into execution plan.
   * @returns {{ groups: array<array<string>>, nodeMap: object }}
   *   groups: each group is an array of node IDs that can run in parallel
   */
  plan(flowDef) {
    const { nodes, edges } = flowDef;
    const nodeMap = {};
    for (const n of nodes) {
      nodeMap[n.id] = n;
    }

    // Build adjacency: nodeId -> { dependencies: [], dependents: [] }
    const adj = {};
    for (const n of nodes) {
      adj[n.id] = { node: n, dependencies: [], dependents: [] };
    }
    for (const e of edges) {
      adj[e.to].dependencies.push(e.from);
      adj[e.from].dependents.push(e.to);
    }

    // Topological sort into groups (Kahn's algorithm)
    const inDegree = {};
    for (const n of nodes) {
      inDegree[n.id] = adj[n.id].dependencies.length;
    }

    const groups = [];
    const remaining = new Set(nodes.map(n => n.id));

    while (remaining.size > 0) {
      const group = [];
      for (const id of remaining) {
        if (inDegree[id] === 0) {
          group.push(id);
        }
      }

      if (group.length === 0) {
        throw new Error(`Circular dependency detected. Remaining: ${[...remaining].join(', ')}`);
      }

      groups.push(group);

      for (const id of group) {
        remaining.delete(id);
        for (const dep of adj[id].dependents) {
          inDegree[dep]--;
        }
      }
    }

    return { groups, nodeMap, adj };
  }

  /**
   * Build a flat execution plan from groups (for simple view display).
   */
  flattenPlan(plan) {
    return plan.groups.map((group, idx) => ({
      phase: idx,
      parallel: group.length > 1,
      nodes: group.map(id => plan.nodeMap[id])
    }));
  }
}

module.exports = WorkflowEngine;

/**
 * Agent Pool — manages parallel execution of workflow nodes.
 * Each node type maps to an executor function.
 * Broadcasts progress via WebSocket callback.

class AgentPool {
  constructor(options = {}) {
    this.vaultReader = options.vaultReader;
    this.configManager = options.configManager;
    this.apiProxy = options.apiProxy;
    this.skillBridge = options.skillBridge;
    this.onProgress = options.onProgress || (() => {});
    this.running = new Map(); // taskId -> { status, result }
  }

  /**
   * Execute a workflow plan.
   * @param {object} plan - from WorkflowEngine.plan()
   * @param {object} context - { vaultPath, chapterPath, duration, shotCount, style }
   * @returns {Promise<object>} all node results keyed by nodeId
   */
  async execute(plan, context) {
    const results = {};
    const { groups, nodeMap } = plan;

    for (const group of groups) {
      const isParallel = group.length > 1;

      if (isParallel) {
        // Run all nodes in this group concurrently
        const promises = group.map(nodeId =>
          this._runNode(nodeId, nodeMap[nodeId], context, results)
            .then(r => ({ nodeId, result: r, error: null }))
            .catch(e => ({ nodeId, result: null, error: e.message }))
        );
        const groupResults = await Promise.all(promises);
        for (const gr of groupResults) {
          results[gr.nodeId] = gr.error ? { error: gr.error } : gr.result;
        }
      } else {
        // Run single node
        const nodeId = group[0];
        try {
          results[nodeId] = await this._runNode(nodeId, nodeMap[nodeId], context, results);
        } catch (e) {
          results[nodeId] = { error: e.message };
        }
      }
    }

    return results;
  }

  async _runNode(nodeId, node, context, previousResults) {
    this.running.set(nodeId, { status: 'running', startedAt: Date.now() });
    this.onProgress({ nodeId, status: 'running', label: node.label });

    try {
      const result = await this._executeByType(node.type, node.params || {}, context, previousResults);
      this.running.set(nodeId, { status: 'completed', result });
      this.onProgress({ nodeId, status: 'completed', label: node.label, result });
      return result;
    } catch (e) {
      this.running.set(nodeId, { status: 'failed', error: e.message });
      this.onProgress({ nodeId, status: 'failed', label: node.label, error: e.message });
      throw e;
    }
  }

  async _executeByType(type, params, context, previousResults) {
    switch (type) {
      case 'obsidian_read':
        return this._execObsidianRead(params, context);
      case 'paste_story':
        return { content: params.content || context.pastedStory || '' };
      case 'extract_all':
        return this._execExtractAll(params, context, previousResults);
      case 'extract_character':
        return this._execExtractCharacter(params, context, previousResults);
      case 'generate_character_sheet':
        return this._execGenerateImage('character_sheet', params, context);
      case 'generate_turnaround':
        return this._execGenerateImage('turnaround', params, context);
      case 'generate_expressions':
        return this._execGenerateImage('expressions', params, context);
      case 'generate_character':
        return this._execGenerateImage('character_card', params, context);
      case 'generate_scene':
        return this._execGenerateImage('scene_card', params, context);
      case 'generate_storyboard':
        return this._execGenerateImage('storyboard', params, context);
      case 'generate_full_board':
        return this._execGenerateImage('full_board', params, context);
      case 'generate_poster':
        return this._execGenerateImage('poster', params, context);
      case 'generate_moodboard':
        return this._execGenerateImage('moodboard', params, context);
      case 'merge_frames':
        return this._execMergeFrames(params, context, previousResults);
      case 'video_prompt':
        return this._execVideoPrompt(params, context, previousResults);
      case 'call_video_api':
        return this._execCallVideoAPI(params, context, previousResults);
      case 'export':
        return this._execExport(params, context, previousResults);
      default:
        throw new Error(`Unknown node type: ${type}`);
    }

  async _execObsidianRead(params, context) {

      mode,

/**
 * Skill Bridge — interface to Claude Code Skill for AI generation.
 * This module is a communication channel; actual AI calls happen
 * via the Claude Code conversation context.
 */

class SkillBridge {
  /**
   * @param {object} options
   * @param {function} options.onRequest - called when skill action is needed
   */
  constructor(options = {}) {
    this.onRequest = options.onRequest || (() => {});
    this.pendingRequests = new Map();
  }

  /**
   * Request AI generation from a sub-skill.
   * Returns a request ID; actual result arrives via onRequest callback.
   */
  async request(options) {
    const { skill, action, params } = options;
    const requestId = `req_${Date.now()}_${Math.random().toString(36).slice(2, 8)}`;

    this.pendingRequests.set(requestId, {
      skill, action, params,
      status: 'pending',
      createdAt: Date.now()
    });

    // The actual AI call is triggered by the server's route handler
    // which communicates with the Claude Code context
    this.onRequest({ requestId, skill, action, params });

    return requestId;
  }

  resolve(requestId, result) {
    const req = this.pendingRequests.get(requestId);
    if (req) {
      req.status = 'completed';
      req.result = result;
      req.completedAt = Date.now();
    }
    return req;
  }

  getStatus(requestId) {
    return this.pendingRequests.get(requestId) || null;
  }
}

module.exports = SkillBridge;

    } else {

    }

      ...data.flows.filter(f => f.category === 'custom')

        ${flow.description ? `<div class="flow-card-desc">${flow.description}</div>` : ''}
      `;

    this.currentFlow = flow;
    this.state = 'pipeline';
    this.render();

    // Subscribe to progress
    this.unsubscribe = onProgress((data) => {
      if (data.workflowId === this.workflowId) {
        this.nodeStatus[data.nodeId] = data;
        this.updateStepStatus(data.nodeId, data);
      }
    });

    // Start workflow
    try {
      const result = await api.startWorkflow({
        flowFile: flow.flowFile,
        chapterPath: flow.chapterPath || '',
        duration: flow.duration || 15,
        shotCount: flow.shotCount || 7
      });
      this.workflowId = result.workflowId;
      this.renderPipelineSteps(result.plan);
    } catch (e) {
      this.showError(e.message);
    }
  }

    this.container.innerHTML = `
      <div class="pipeline-header">
        <button class="btn-back" onclick="simpleView.backToCards()">← 返回</button>
        <h2>${this.currentFlow?.name || '流程'}</h2>

        arrow.className = 'step-arrow';
        arrow.innerHTML = '▶';
        stepsWrapper.appendChild(arrow);
      }

      // Phase group

      phaseGroup.className = `phase-group ${phase.parallel ? 'parallel' : ''}`;

      phase.nodes.forEach(node => {
        const step = document.createElement('div');
        step.className = 'step-card';
        step.id = `step-${node.id}`;
        step.dataset.nodeId = node.id;
        step.innerHTML = `
          <div class="step-status-indicator" id="status-${node.id}">⏳</div>
          <div class="step-label">${node.label || node.id}</div>
          <div class="step-preview" id="preview-${node.id}"></div>
        `;
        phaseGroup.appendChild(step);
      });

      stepsWrapper.appendChild(phaseGroup);
    });

    container.innerHTML = '';
    container.appendChild(stepsWrapper);

    const indicator = document.getElementById(`status-${nodeId}`);

        card.className = `step-card status-${data.status}`;
      }
    }

    if (preview && data.result) {
      if (data.result.url) {
        preview.innerHTML = `<img src="${data.result.url}" class="step-thumb" />`;
      } else if (data.result.videoUrl) {
        preview.innerHTML = `<video src="${data.result.videoUrl}" class="step-thumb" controls></video>`;
      }
    }
  }

    this.state = 'cards';
    this.currentFlow = null;
    this.workflowId = null;

// studio/ui/js/canvas.js
class FlowCanvas {
  constructor(svgElement) {
    this.svg = svgElement;
    this.nodes = [];
    this.edges = [];
    this.dragging = null;
    this.connecting = null; // { fromNodeId, tempLine }
    this._setupEvents();
  }

  addNode(nodeDef, x, y) {
    const g = document.createElementNS('http://www.w3.org/2000/svg', 'g');
    g.setAttribute('transform', `translate(${x},${y})`);
    g.setAttribute('data-node-id', nodeDef.id);
    g.classList.add('canvas-node');

    // Node body
    const rect = document.createElementNS('http://www.w3.org/2000/svg', 'rect');
    rect.setAttribute('width', '140');
    rect.setAttribute('height', '56');
    rect.setAttribute('rx', '8');
    rect.classList.add('node-body');
    g.appendChild(rect);

    // Node label
    const text = document.createElementNS('http://www.w3.org/2000/svg', 'text');
    text.setAttribute('x', '70');
    text.setAttribute('y', '32');
    text.setAttribute('text-anchor', 'middle');
    text.textContent = nodeDef.label || nodeDef.id;
    text.classList.add('node-label');
    g.appendChild(text);

    // Output port (right side)
    const outPort = document.createElementNS('http://www.w3.org/2000/svg', 'circle');
    outPort.setAttribute('cx', '140');
    outPort.setAttribute('cy', '28');
    outPort.setAttribute('r', '6');
    outPort.classList.add('port', 'port-out');
    outPort.dataset.port = 'out';
    outPort.dataset.nodeId = nodeDef.id;
    g.appendChild(outPort);

    // Input port (left side)
    const inPort = document.createElementNS('http://www.w3.org/2000/svg', 'circle');
    inPort.setAttribute('cx', '0');
    inPort.setAttribute('cy', '28');
    inPort.setAttribute('r', '6');
    inPort.classList.add('port', 'port-in');
    inPort.dataset.port = 'in';
    inPort.dataset.nodeId = nodeDef.id;
    g.appendChild(inPort);

    // Status indicator
    const statusDot = document.createElementNS('http://www.w3.org/2000/svg', 'circle');
    statusDot.setAttribute('cx', '130');
    statusDot.setAttribute('cy', '10');
    statusDot.setAttribute('r', '4');
    statusDot.classList.add('status-dot', 'status-pending');
    g.appendChild(statusDot);

    this.svg.appendChild(g);

    this.nodes.push({ id: nodeDef.id, def: nodeDef, g, x, y });
    return g;
  }

  addEdge(fromNodeId, toNodeId) {
    // Avoid duplicates
    if (this.edges.find(e => e.from === fromNodeId && e.to === toNodeId)) return;

    const line = document.createElementNS('http://www.w3.org/2000/svg', 'line');
    line.classList.add('edge-line');
    line.dataset.from = fromNodeId;
    line.dataset.to = toNodeId;
    this.svg.insertBefore(line, this.svg.firstChild); // edges behind nodes

    this.edges.push({ from: fromNodeId, to: toNodeId, line });
    this._updateEdgePosition(line, fromNodeId, toNodeId);
  }

  _updateEdgePosition(line, fromId, toId) {
    const fromNode = this.nodes.find(n => n.id === fromId);
    const toNode = this.nodes.find(n => n.id === toId);
    if (!fromNode || !toNode) return;

    line.setAttribute('x1', fromNode.x + 140);
    line.setAttribute('y1', fromNode.y + 28);
    line.setAttribute('x2', toNode.x);
    line.setAttribute('y2', toNode.y + 28);
  }

  _setupEvents() {
    this.svg.addEventListener('mousedown', (e) => this._onMouseDown(e));
    this.svg.addEventListener('mousemove', (e) => this._onMouseMove(e));
    this.svg.addEventListener('mouseup', (e) => this._onMouseUp(e));
  }

  _onMouseDown(e) {
    const port = e.target.closest('.port');
    if (port) {
      // Start connecting
      const nodeId = port.dataset.nodeId;
      if (port.classList.contains('port-out')) {
        this.connecting = { fromNodeId: nodeId };
        e.stopPropagation();
        return;
      }
    }

    const node = e.target.closest('.canvas-node');
    if (node) {
      this.dragging = {
        nodeId: node.dataset.nodeId,
        offsetX: e.clientX - parseFloat(node.getAttribute('transform').match(/translate\(([^,]+),([^)]+)\)/)[1]),
        offsetY: e.clientY - parseFloat(node.getAttribute('transform').match(/translate\(([^,]+),([^)]+)\)/)[2])
      };
    }
  }

  _onMouseMove(e) {
    if (this.dragging) {
      const node = this.svg.querySelector(`[data-node-id="${this.dragging.nodeId}"]`);
      if (!node) return;
      const x = e.clientX - this.dragging.offsetX;
      const y = e.clientY - this.dragging.offsetY;
      node.setAttribute('transform', `translate(${x},${y})`);

      // Update connected edges
      const n = this.nodes.find(n => n.id === this.dragging.nodeId);
      if (n) { n.x = x; n.y = y; }
      this.edges.forEach(edge => {
        this._updateEdgePosition(edge.line, edge.from, edge.to);
      });
    }
  }

  _onMouseUp(e) {
    if (this.connecting) {
      const port = e.target.closest('.port-in');
      if (port) {
        const toNodeId = port.dataset.nodeId;
        if (toNodeId !== this.connecting.fromNodeId) {
          this.addEdge(this.connecting.fromNodeId, toNodeId);
        }
      }
      this.connecting = null;
    }
    this.dragging = null;
  }

  // Load from flow definition
  loadFlow(flowDef) {
    this.nodes = [];
    this.edges = [];
    this.svg.innerHTML = '';

    // Layout nodes in a grid
    const nodeMap = {};
    flowDef.nodes.forEach((node, idx) => {
      const col = idx % 3;
      const row = Math.floor(idx / 3);
      const g = this.addNode(node, col * 200 + 40, row * 100 + 40);
      nodeMap[node.id] = { x: col * 200 + 40, y: row * 100 + 40 };
    });

    flowDef.edges.forEach(edge => {
      this.addEdge(edge.from, edge.to);
    });
  }

  // Export current canvas as flow definition
  exportFlow() {
    const nodeDefs = this.nodes.map(n => ({
      id: n.id,
      type: n.def.type,

class ProView {
  constructor(container, svgContainer, libraryContainer) {

    this.canvas = new FlowCanvas(svgContainer);
    this.libraryContainer = libraryContainer;
    this.currentFlowName = null;
  }

    this._setupDropTarget();
  }

  _setupDropTarget() {
    this.canvas.svg.addEventListener('dragover', (e) => e.preventDefault());
    this.canvas.svg.addEventListener('drop', (e) => {
      e.preventDefault();
      const data = JSON.parse(e.dataTransfer.getData('application/json'));
      const rect = this.canvas.svg.getBoundingClientRect();
      const x = e.clientX - rect.left;
      const y = e.clientY - rect.top;
      this.canvas.addNode(data, x, y);
    });
  }

  loadPresetFlow(flowDef) {
    this.canvas.loadFlow(flowDef);
  }

  exportFlow() {
    return this.canvas.exportFlow();
  }

  async saveAsCustomWorkflow(name) {
    const flowDef = this.exportFlow();
    flowDef.name = name;

    const config = await api.getConfig();
    config.custom_workflows = config.custom_workflows || [];
    config.custom_workflows.push({
      name,
      icon: '🔧',
      flowDef,
      flowFile: `custom_${Date.now()}`
    });
    await api.updateConfig({ custom_workflows: config.custom_workflows });
  }

  async startExecution(platform) {
    const flowDef = this.exportFlow();
    const tempFlowFile = `temp_${Date.now()}`;

    // Save temp flow and start

      const result = await api.startWorkflow({
        flowFile: tempFlowFile,
        platform
      });
      return result;
    } catch (e) {
      console.error('Execution failed:', e);
      throw e;
    }
  }

        </section>

        <section class="config-section">
          <h3>偏好设置</h3>

        </section>

      </div>
    `;
  }

  _renderKeyFields(sectionKey, data, keys) {
    return keys.map(k => {
      const val = data?.[k]?.key || '';
      const masked = val ? '••••••••' : '';
      return `
        <div class="config-field">
          <label>${k}</label>
          <input type="password" data-section="${sectionKey}" data-key="${k}" data-subkey="key"
                 placeholder="${masked || '未配置'}" />
          ${k === 'seedance' ? '<input type="text" data-section="' + sectionKey + '" data-key="' + k + '" data-subkey="endpoint" placeholder="自定义 endpoint（可选）" />' : ''}
        </div>
      `;
    }).join('');
  }

  _renderDefaults(defaults) {
    const fields = [
      { key: 'aspect_ratio', label: '默认画幅' },
      { key: 'duration', label: '默认时长（秒）' },
      { key: 'shots_per_15s', label: '每15秒分镜数' },
      { key: 'style', label: '默认风格' },
      { key: 'output_dir', label: '产物目录' }
    ];
    return fields.map(f => `
      <div class="config-field">
        <label>${f.label}</label>
        <input type="text" data-section="defaults" data-key="${f.key}" value="${defaults?.[f.key] || ''}" />
      </div>
    `).join('');
  }

  _renderPreferences(prefs) {
    return `
      <div class="config-field">
        <label>输出语言</label>
        <select data-section="preferences" data-key="language">
          <option value="zh" ${prefs?.language === 'zh' ? 'selected' : ''}>中文</option>
          <option value="en" ${prefs?.language === 'en' ? 'selected' : ''}>English</option>
        </select>
      </div>
    `;
  }

  async save() {
    const config = {};

    this.container.querySelectorAll('input, select').forEach(el => {
      const section = el.dataset.section;
      const key = el.dataset.key;
      const subkey = el.dataset.subkey;
      if (!section || !key) return;

      config[section] = config[section] || {};
      if (subkey) {
        config[section][key] = config[section][key] || {};
        config[section][key][subkey] = el.value;
      } else {
        config[section][key] = el.value;
      }
    });

    try {
      await api.updateConfig(config);
      alert('配置已保存 ✅');
    } catch (e) {
      alert('保存失败: ' + e.message);
    }
  }
}
